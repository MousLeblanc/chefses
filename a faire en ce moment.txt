const ErrorService = {
  // Types d'erreurs
  errorTypes: {
    NETWORK: 'network',
    API: 'api',
    VALIDATION: 'validation',
    AUTHENTICATION: 'auth',
    PERMISSION: 'permission',
    NOT_FOUND: 'not_found',
    UNKNOWN: 'unknown'
  },
  
  // G√©rer les erreurs selon leur type
  handleError(error, context = '') {
    let errorType = this.errorTypes.UNKNOWN;
    let errorMessage = 'Une erreur inattendue est survenue.';
    
    // D√©terminer le type d'erreur
    if (error.name === 'TypeError' && error.message.includes('fetch')) {
      errorType = this.errorTypes.NETWORK;
      errorMessage = 'Erreur de connexion au serveur. V√©rifiez votre connexion Internet.';
    } else if (error.status) {
      // Erreur API avec code de statut
      switch (error.status) {
        case 400:
          errorType = this.errorTypes.VALIDATION;
          errorMessage = 'Les donn√©es envoy√©es sont incorrectes.';
          break;
        case 401:
          errorType = this.errorTypes.AUTHENTICATION;
          errorMessage = 'Votre session a expir√©. Veuillez vous reconnecter.';
          this.redirectToLogin();
          break;
        case 403:
          errorType = this.errorTypes.PERMISSION;
          errorMessage = 'Vous n\'avez pas les permissions n√©cessaires.';
          break;
        case 404:
          errorType = this.errorTypes.NOT_FOUND;
          errorMessage = 'La ressource demand√©e n\'existe pas.';
          break;
        case 500:
          errorType = this.errorTypes.API;
          errorMessage = 'Erreur serveur. Veuillez r√©essayer plus tard.';
          break;
        default:
          errorType = this.errorTypes.API;
          errorMessage = `Erreur API: ${error.status}`;
      }
    }
    
    // Logging de l'erreur
    console.error(`[${context}] ${errorType}: ${errorMessage}`, error);
    
    // Afficher une notification utilisateur
    NotificationSystem.error(errorMessage);
    
    // Tracking d'erreur (√† impl√©menter avec un service comme Sentry)
    // this.trackError(errorType, errorMessage, context, error);
    
    return { type: errorType, message: errorMessage };
  },
  
  // Rediriger vers la page de connexion si n√©cessaire
  redirectToLogin() {
    localStorage.removeItem('chaif-ses-authenticated');
    localStorage.removeItem('chaif-ses-token');
    window.location.href = 'login.html?redirect=timeout';
  },
  
  // Valider les donn√©es de formulaire
  validateForm(formData, rules) {
    const errors = {};
    
    for (const field in rules) {
      if (rules.hasOwnProperty(field)) {
        const value = formData[field];
        const fieldRules = rules[field];
        
        // V√©rifier si requis
        if (fieldRules.required && (!value || value.trim() === '')) {
          errors[field] = 'Ce champ est requis';
          continue;
        }
        
        // V√©rifier la longueur minimale
        if (fieldRules.minLength && value && value.length < fieldRules.minLength) {
          errors[field] = `Ce champ doit contenir au moins ${fieldRules.minLength} caract√®res`;
          continue;
        }
        
        // V√©rifier le pattern (ex: email, num√©ro, etc.)
        if (fieldRules.pattern && value && !fieldRules.pattern.test(value)) {
          errors[field] = fieldRules.message || 'Format invalide';
          continue;
        }
        
        // V√©rifier les valeurs min/max pour les nombres
        if (fieldRules.type === 'number') {
          const numValue = Number(value);
          if (isNaN(numValue)) {
            errors[field] = 'Ce champ doit √™tre un nombre';
            continue;
          }
          if (fieldRules.min !== undefined && numValue < fieldRules.min) {
            errors[field] = `La valeur minimale est ${fieldRules.min}`;
            continue;
          }
          if (fieldRules.max !== undefined && numValue > fieldRules.max) {
            errors[field] = `La valeur maximale est ${fieldRules.max}`;
            continue;
          }
        }
      }
    }
    
    return Object.keys(errors).length === 0 ? null : errors;
  }
};
